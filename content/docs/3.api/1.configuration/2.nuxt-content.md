The default nuxt content config can be overridden in `nuxt.config.ts`.

```ts [nuxt.config.ts]
export default defineNuxtConfig({
  // Here
  content: {
    // ...
  },

  compatibilityDate: '2026-01-30',
});
```

## Syntax Highlighting

### Theme

```ts [nuxt.config.ts]
export default defineNuxtConfig({
  content: {
    build: {
      markdown: {
        highlight: {
          theme: {
            // Default theme
            default: 'github-light',
            // Theme used in dark-mode
            dark: 'github-dark',
          },
        },
      },
    },
  },
});
```

:ct-read-more{to="https://shiki.style/themes"}

### Languages

The default highlighted languages are: `['json', 'js', 'ts', 'html', 'css', 'vue', 'diff', 'shell', 'markdown', 'mdc', 'yaml', 'bash', 'ini', 'dotenv']`{lang="js"}.

If you need to highlight other languages, add them in your `nuxt.config.ts`.

```ts [nuxt.config.ts]
export default defineNuxtConfig({
  content: {
    build: {
      markdown: {
        highlight: {
          preload: ['c', 'cpp'],
        },
      },
    },
  },
});
```

:ct-read-more{to="https://shiki.style/languages"}

## Default Config

```ts [nuxt.config.ts]
export default defineNuxtConfig({
  content: {
    build: {
      markdown: {
        highlight: {
          theme: {
            default: 'github-light',
            dark: 'github-dark',
          },
          preload: ['json', 'js', 'ts', 'html', 'css', 'vue', 'diff', 'shell', 'markdown', 'mdc', 'yaml', 'bash', 'ini', 'dotenv'],
        },
      },
    },
  },
});
```
